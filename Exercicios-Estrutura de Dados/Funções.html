                Lista *novo, *p=l;

                novo = (Lista*)malloc(sizeof(Lista));

                novo->info = v;

                novo->prox = NULL;

               

                if(p!=NULL){

                               while(p->prox!=NULL)

                                               p = p->prox;

                               p->prox = novo;

                }

                else

                               l = novo;                             

                return l;

}

                while(p!=NULL){

                               p = p->prox;

                }

}

void imprimirRec(Lista *l){

                if(l!=NULL){

                               printf("%x [%d | %x] -> ",l,l->info,l->prox);

                               imprimirRec(l->prox);

                }                                             

}

                               printf("Lista vazia.");

                }

                else{

                                               ant = p;

                               }

                              

                               if(ant==NULL){

                                               l = p->prox;                                       

                               }

                               else{

                                               ant->prox = p->prox;                                     

                               }

                               free(p);

                               return l;

                }

                              

}

                else if(l2 == NULL) {

#include <stdio.h>

#include <stdlib.h>

 

#define MAX 5

 

struct fila{

                int ini;

                int fim;

                int qtd;

                int vet[MAX];

};

typedef struct fila Fila;

 

Fila* cria(){

                Fila *novo;

                novo = (Fila*)malloc(sizeof(Fila));

                novo->ini = 0;

                novo->fim = 0;

                novo->qtd = 0;

                return novo;

}

 

int incr(int i){

                return (i+1)%MAX;

}

 

void inserir(Fila *f, int valor){

                if(f->qtd != MAX){

                               f->vet[f->fim] = valor;

                               f->fim = incr(f->fim);

                               f->qtd++;

                }

               

                else{

                               printf("Nao ha espaco na fila!");

                }

}

 

void retirar(Fila *f){

                if(f->qtd==0){

                               printf("Fila vazia");

                }

               

                else{

                               f->ini = incr(f->ini);

                               f->qtd--;

                }

}

 

//Função utilizada para inverter os valores da fila

void reverso(Fila *f) {

    if (f == NULL) {

        printf("A fila esta vazia.\n");

        return;

    }

 

    int *temp = (int *)malloc(f->qtd * sizeof(int));

    int i = f->ini;

 

    int cont;

   

    for (cont = 0; cont < f->qtd; cont++) {

        temp[cont] = f->vet[i];

        i = (i + 1) % MAX;

    }

 

    i = f->ini;

    for (cont = f->qtd - 1; cont >= 0; cont--) {

        f->vet[i] = temp[cont];

        i = (i + 1) % MAX;

    }

 

    free(temp);

}

 

//Função utilizada para ordenar os valores da fila

void ordenar(Fila *f) {

    int i, j, temp;

 

    for (i = 0; i < f->qtd - 1; i++) {

        for (j = i + 1; j < f->qtd; j++) {

            if (f->vet[i] > f->vet[j]) {

                temp = f->vet[i];

                f->vet[i] = f->vet[j];

                f->vet[j] = temp;

            }

        }

    }

}

 

//Função utilizada para calcular o maior, menor e media de todos os valores da fila

void calcular(Fila *f, int *maior, int *menor, float *media) {

    if (f == NULL) {

        printf("Erro: a fila está vazia\n");

        return;

    }

 

    *maior = *menor = f->vet[f->ini];

    *media = f->vet[f->ini];

    int quantidadeElementos = 1;

 

                int i;

               

    for (i = 1; i < f->qtd; i++) {

        int valor = f->vet[(f->ini + i) % MAX];

 

        // Atualiza o maior valor

        if (valor > *maior) {

            *maior = valor;

        }

 

        // Atualiza o menor valor

        if (valor < *menor) {

            *menor = valor;

        }

 

        // Acumula os valores para calcular a média no final

        *media += valor;

        quantidadeElementos++;

    }

 

    // Calcula a média

    *media /= quantidadeElementos;

}

 

//Função utilizada para excluir valores negativos da fila

void retirarNegativos(Fila *f) {

    if (f->qtd == 0) {

        printf("Fila vazia\n");

        return;

    }

 

    Fila *temp = cria();

    while (f->qtd > 0) {

        int valor = f->vet[f->ini];

        retirar(f);

 

        if (valor >= 0) {

            inserir(temp, valor);

        }

    }

 

 

    while (temp->qtd > 0) {

        int valor = temp->vet[temp->ini];

        retirar(temp);

        inserir(f, valor);

    }

 

    free(temp);

}

                int contador = 0;

                                               }

               }

                }

 

                return contador;

}

//Função utilizada para copiar uma fila e retorna a copia

Fila* copia(Fila* original) {

    Fila* copia = cria();

 

    if (original->qtd == 0) {

        return copia;

    }

 

    int i, j;

    for (i = 0, j = original->ini; i < original->qtd; i++, j = incr(j)) {

        copia->vet[i] = original->vet[j];

    }

 

    copia->ini = 0;

    copia->fim = (original->qtd - 1 + MAX) % MAX;

    copia->qtd = original->qtd;

 

    return copia;

}

//Função utilizada para inverter os valores da fila

void reverso(Fila* fila) {

    No* anterior = NULL;

    No* atual = fila->ini;

    No* proximo = NULL;

 

    while (atual != NULL) {

        proximo = atual->prox;

        atual->prox = anterior;

        anterior = atual;

        atual = proximo;

    }

 

    fila->ini = anterior;

    fila->fim = fila->ini; // O que era o início agora é o fim

}

 

 

//Função utilizada para ordenar os valores da fila

void ordenar(Fila* fila) {

    int trocado;

    No *atual, *proximo = NULL;

 

    do {

        trocado = 0;

        atual = fila->ini;

 

        while (atual->prox != proximo) {

            if (atual->info > atual->prox->info) {

                // Trocar os elementos se estiverem fora de ordem

                int temp = atual->info;

                atual->info = atual->prox->info;

                atual->prox->info = temp;

                trocado = 1;

            }

            atual = atual->prox;

        }

 

        proximo = atual;

 

    } while (trocado);

}

 

//Função utilizada para calcular o maior, menor e media de todos os valores da fila

void calcular(Fila *fila, int* maior, int* menor, float* media) {

    if (fila->ini == NULL) {

        printf("Erro: a fila está vazia\n");

        exit(EXIT_FAILURE);

    }

 

    No* atual = fila->ini;

    *maior = *menor = atual->info;

    *media = atual->info;

    int quantidadeElementos = 1;

 

    while (atual->prox != NULL) {

        atual = atual->prox;

 

        // Atualiza o maior valor

        if (atual->info > *maior) {

            *maior = atual->info;

        }

 

        // Atualiza o menor valor

        if (atual->info < *menor) {

            *menor = atual->info;

        }

 

        // Acumula os valores para calcular a média no final

        *media += atual->info;

        quantidadeElementos++;

    }

 

    // Calcula a média

    *media /= quantidadeElementos;

}

//Funcao para inserir um no na pilha utilizando de parametro a pilha e o valor que serah inserido

void push(Pilha* p, int valor) {

    No* novo = (No*)malloc(sizeof(No));

    if (novo == NULL) {

        printf("Erro ao alocar memória para o nó.\n");

        exit(0);

    }

    novo->info = valor;

    novo->prox = p->topo;

    p->topo = novo;

}

 

//Funcao para verificar se a pilha estah vazia utilizando de parametro a pilha

int pilhaVazia(Pilha* p) {

    return (p->topo == NULL);

}

 

//Funcao para retirar o ultimo no da pilha utilizando de parametro a pilha

int pop(Pilha* p) {

    if (pilhaVazia(p)){

        printf("Erro: a pilha está vazia.\n");

        exit(0);

    }

    No* temp = p->topo;

    int valor = temp->info;

    p->topo = temp->prox;

    free(temp);

    return valor;

}

 

//Funcao para ordenar a pilha de forma crescente utilizando de parametro a pilha

void ordenarPilha(Pilha* p) {

    Pilha* pilhaTemp = criaPilha();

 

    while (!pilhaVazia(p)) {

        int valor = pop(p);

 

        while (!pilhaVazia(pilhaTemp) && pilhaTemp->topo->info > valor) {

            push(p, pop(pilhaTemp));

        }

 

        push(pilhaTemp, valor);

    }

 

    while (!pilhaVazia(pilhaTemp)) {

        push(p, pop(pilhaTemp));

    }

 

    free(pilhaTemp);

}

 

//Funcao para contar quantos elementos uma pilha tem utilizando de parametro a pilha

int contarElementosPilha(Pilha* p){

    int contador = 0;

    No* atual = p->topo;

 

    while (atual != NULL) {

        contador++;

        atual = atual->prox;

    }

 

    return contador;

}

 

//Funcao para retornar estatisticas sobre a pilha utilizando de parametro a pilha

Result estatisticasPilha(Pilha* p){

    if (pilhaVazia(p)) {

        printf("Erro: a pilha está vazia.\n");

        exit(0);

    }

 

    int soma = 0;

    Result resultado;

    resultado.menor = p->topo->info;

    resultado.maior = p->topo->info;

 

    No* atual = p->topo;

 

    while (atual != NULL) {

        int valor = atual->info;

 

        if (valor < resultado.menor) {

            resultado.menor = valor;

        }

 

        if (valor > resultado.maior) {

            resultado.maior = valor;

        }

 

        soma += valor;

 

        atual = atual->prox;

    }

 

    resultado.media = (float)soma / contarElementosPilha(p);

 

    return resultado;

}

}

    int c = contarElementosPilha(p);

 

    int vet[c];

        vet[i++] = pop(p);

    }

 

    for(int j = 0; j < c; j++) {

        push(p, vet[j]);

    }

 

    return p;

}

 

//Funcao para inserir n valores da sequencia de fibonnaci utilizando de parametro a pilha e a quant

Pilha* fibonacciPilha(int n){

    Pilha* novo = criaPilha();

 

    int vet[n];

    int i, atual = 1, anterior = 0;

 

    for(i = 0; i < n; i++){

        if(i < 2){

            if(i < 1){

                vet[i] = 0;

            }

            else{

                vet[i] = 1;

            }

        }

        else{

            vet[i] = anterior + atual;

            anterior = atual;

            atual = vet[i];

        }

    }

 

    for(int j = 0; j < n; j++) {

        push(novo, vet[j]);

    }

 

    return novo;

}
